#### 1. LIST ALL THE DEPTNAME AND LOC OF ALL THE SALESMAN MANAGER-MANAGER'S

    SELECT DNAME,LOC
    FROM DEPT
    WHERE DEPTNO IN(SELECT DEPTNO
    FROM EMP
    WHERE EMPNO IN(SELECT MGR
    FROM EMP
    WHERE EMPNO IN(SELECT MGR
    FROM EMP
    WHERE JOB='SALESMAN')));

#### 2.DISPLAY THE LAST EMPLOYEE RECORD( HIRED LAST) WITH 25% HIKE IN SALARY.

    SELECT ENAME,SAL,HIREDATE,SAL*1.25 AS NEW_SALARY
    FROM EMP
    WHERE HIREDATE=(SELECT MAX(HIREDATE)
                    FROM EMP);

#### 3.DISPLAY DNAME, LOC, DEPTNO OF EMPLOYEES WHO HAS SAME REPORTING MANAGER?----

    SELECT DNAME,LOC,DEPTNO
    FROM DEPT
    WHERE DEPTNO IN(SELECT DEPTNO
    FROM EMP

    FROM EMP

    HAVING COUNT(*)>1));

#### 5. DISPLAY ALL THE EMPLOYEES WHOSE SALARY IS GREATER THAN AVG SAL OF DEPARTMENT 20

    FROM EMP

    FROM EMP

#### 8.WAQTD THE NAMES OF THE EMPLOYEE WHO HAVE SAME REPORTING MANAGER AND EARN SAL MORE THAN THE FIRST HIRED EMPLOYEE-

    FROM EMP

#### 10. Find the employees who earn the highest salary in the department where the minimum salary is greater than 1500.

    SELECT ENAME
    FROM EMP
    WHERE SAL IN(SELECT MAX(SAL)
                  FROM EMP
                  WHERE DEPTNO IN(SELECT DEPTNO
                                  FROM EMP
                                  GROUP BY DEPTNO
                                  HAVING MIN(SAL)>1500)
                  GROUP BY DEPTNO);


    SELECT *
    FROM EMP
    WHERE(SAL, DEPTNO) IN (SELECT MAX(SAL), DEPTNO
                            FROM EMP
                            GROUP BY DEPTNO
                            HAVING MIN(SAL)>1500);
